\begin{equation}
  \nu = a y^3 -2.25 a y^2 + cy
\end{equation}

\begin{equation}
  \frac{d\nu}{dy} = 3 a y^2 - 4.5 a y + c
\end{equation}

\begin{equation}
  \frac{d^2 \nu}{dy^2} = 6 a y - 4.5 a
\end{equation}
